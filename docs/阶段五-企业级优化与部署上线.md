## 5.2 安全与限流防护

### API 限流

```javascript
// 限流中间件
import rateLimit from 'express-rate-limit'

// 全局限流
const globalLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,  // 15 分钟
  max: 100,                    // 每个 IP 最多 100 次请求
  message: {
    error: '请求过于频繁，请稍后重试'
  }
})

// 聊天接口限流（更严格）
const chatLimiter = rateLimit({
  windowMs: 60 * 1000,  // 1 分钟
  max: 20,               // 每个 IP 最多 20 次
  message: {
    error: '聊天请求过于频繁，请稍后重试'
  }
})

app.use('/api/', globalLimiter)
app.use('/api/chat/', chatLimiter)
```

### API Key 鉴权

```javascript
// API Key 验证中间件
function apiKeyAuth(req, res, next) {
  const apiKey = req.headers['x-api-key']

  if (!apiKey) {
    return res.status(401).json({
      error: '缺少 API Key'
    })
  }

  // 验证 API Key
  const validKeys = process.env.API_KEYS?.split(',') || []
  if (!validKeys.includes(apiKey)) {
    return res.status(403).json({
      error: 'API Key 无效'
    })
  }

  next()
}

// 使用
app.use('/api/', apiKeyAuth)
```

### 输入验证

```javascript
// 输入验证（使用 Joi）
import Joi from 'joi'

// 聊天消息验证
const chatSchema = Joi.object({
  message: Joi.string()
    .min(1)
    .max(1000)
    .required()
    .messages({
      'string.min': '消息不能为空',
      'string.max': '消息过长，最多 1000 个字符',
      'any.required': '消息内容是必填项'
    }),

  history: Joi.array()
    .max(50)
    .optional()
})

// 中间件
function validateChat(req, res, next) {
  const { error } = chatSchema.validate(req.body)
  if (error) {
    return res.status(400).json({
      error: error.details[0].message
    })
  }
  next()
}
```

## 5.3 容器化部署

### Dockerfile

```dockerfile
# 前端构建阶段
FROM node:18-alpine AS frontend-builder

WORKDIR /app/frontend

# 复制依赖文件
COPY frontend/package*.json ./

# 安装依赖
RUN npm ci --only=production

# 复制源码
COPY frontend/ .

# 构建
RUN npm run build

# 后端构建阶段
FROM node:18-alpine AS backend-builder

WORKDIR /app/backend

# 复制依赖文件
COPY backend/package*.json ./

# 安装依赖
RUN npm ci --only=production

# 复制源码
COPY backend/ .

# 生产阶段
FROM node:18-alpine

WORKDIR /app

# 复制前端构建产物
COPY --from=frontend-builder /app/frontend/dist ./frontend/dist

# 复制后端
COPY --from=backend-builder /app/backend ./backend

# 安装生产依赖
RUN cd backend && npm ci --only=production

# 设置环境变量
ENV NODE_ENV=production
ENV PORT=8001

# 暴露端口
EXPOSE 8001

# 启动命令
CMD ["node", "backend/src/index.js"]
```

### Nginx 配置

```nginx
# nginx.conf
events {
    worker_connections 1024;
}

http {
    # Gzip 压缩
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;

    # 上游服务
    upstream webgis_backend {
        server webgis-agent:8001;
    }

    server {
        listen 80;
        server_name your-domain.com;

        # 前端静态文件
        location / {
            root /app/frontend/dist;
            try_files $uri $uri/ /index.html;
        }

        # API 代理
        location /api/ {
            proxy_pass http://webgis_backend;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

            # SSE 支持
            proxy_buffering off;
            proxy_cache off;
        }

        # Cesium 静态资源
        location /cesium/ {
            alias /app/frontend/dist/cesium/;
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
    }
}
```

## 5.4 CI/CD 流水线

### GitHub Actions

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'

      - name: Install dependencies
        run: |
          cd frontend && npm ci
          cd ../backend && npm ci

      - name: Run tests
        run: |
          cd frontend && npm test
          cd ../backend && npm test

      - name: Build
        run: |
          cd frontend && npm run build

      - name: Build Docker image
        run: docker build -t webgis-agent:${{ github.sha }} .

      - name: Push to registry
        run: |
          echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker tag webgis-agent:${{ github.sha }} registry.example.com/webgis-agent:latest
          docker push registry.example.com/webgis-agent:latest

      - name: Deploy to server
        uses: appleboy/ssh-action@v0.1.6
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: ${{ secrets.SERVER_USERNAME }}
          key: ${{ secrets.SERVER_SSH_KEY }}
          script: |
            cd /opt/webgis-agent
            docker-compose pull
            docker-compose up -d
```

## 5.5 业务可扩展性预留

### 工具插件系统

```javascript
// 插件接口定义
interface ToolPlugin {
  name: string
  version: string
  description: string
  tools: ToolDefinition[]
  init(context: PluginContext): Promise<void>
  destroy(): Promise<void>
}

// 插件管理器
class PluginManager {
  private plugins: Map<string, ToolPlugin>

  async register(plugin: ToolPlugin) {
    await plugin.init(this.context)
    this.plugins.set(plugin.name, plugin)
  }

  async unregister(name: string) {
    const plugin = this.plugins.get(name)
    if (plugin) {
      await plugin.destroy()
      this.plugins.delete(name)
    }
  }

  getTools(): ToolDefinition[] {
    return Array.from(this.plugins.values())
      .flatMap(p => p.tools)
  }
}
```

### WMS/WFS 图层接入

```javascript
// WMS 图层管理器
class WMSLayerManager {
  constructor(viewer) {
    this.viewer = viewer
    this.layers = new Map()
  }

  // 添加 WMS 图层
  addLayer(name, options) {
    const { url, layers, parameters } = options

    const provider = new Cesium.WebMapServiceImageryProvider({
      url,
      layers,
      parameters: {
        transparent: true,
        format: 'image/png',
        ...parameters
      }
    })

    const layer = this.viewer.imageryLayers.addImageryProvider(provider)
    this.layers.set(name, layer)

    return layer
  }

  // 移除图层
  removeLayer(name) {
    const layer = this.layers.get(name)
    if (layer) {
      this.viewer.imageryLayers.remove(layer)
      this.layers.delete(name)
    }
  }

  // 切换图层可见性
  toggleLayer(name) {
    const layer = this.layers.get(name)
    if (layer) {
      layer.show = !layer.show
    }
  }
}
```
